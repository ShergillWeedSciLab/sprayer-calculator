<script>
const FT_PER_MILE = 5280.0;
const SEC_PER_HOUR = 3600.0;
const SQFT_PER_ACRE = 43560.0;
const ML_TO_GALLON = 0.000264172; // 1 mL = 0.000264172 gallons

function calculate() {
    const nNozzles = parseFloat(document.getElementById("n_nozzles").value);
    const spacingIn = parseFloat(document.getElementById("spacing_in").value);
    const distFt = parseFloat(document.getElementById("calib_distance_ft").value);
    const timeSec = parseFloat(document.getElementById("calib_time_sec").value);
    const volMlNozzle = parseFloat(document.getElementById("volume_ml_per_nozzle").value);

    if (!nNozzles || !spacingIn || !distFt || !timeSec || !volMlNozzle) {
        alert("Please enter all values.");
        return;
    }

    const volGalNozzle = volMlNozzle * ML_TO_GALLON;
    const speedMph = (distFt / timeSec) * (SEC_PER_HOUR / FT_PER_MILE);
    const gpm = volGalNozzle * (60 / timeSec);
    const vol15 = gpm * 0.25;
    const vol15ml = vol15 / ML_TO_GALLON;
    const boomWidthFt = (spacingIn * nNozzles) / 12.0;
    const areaAcres = (distFt * boomWidthFt) / SQFT_PER_ACRE;
    const totalGal15 = vol15 * nNozzles;
    const gpa = (volGalNozzle * nNozzles * (SQFT_PER_ACRE / (distFt * boomWidthFt)));

    document.getElementById("results").style.display = "block";
    document.getElementById("speed").innerText = speedMph.toFixed(2);
    document.getElementById("gpm").innerText = gpm.toFixed(3);
    document.getElementById("vol15").innerText = vol15.toFixed(4);
    document.getElementById("vol15ml").innerText = vol15ml.toFixed(1);
    document.getElementById("total15").innerText = totalGal15.toFixed(4);
    document.getElementById("gpa").innerText = gpa.toFixed(2);
}
</script>
